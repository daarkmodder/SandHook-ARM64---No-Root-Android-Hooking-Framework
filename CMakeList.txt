# CMakeLists.txt
# Build script for SandHook ARM64 Production Framework

cmake_minimum_required(VERSION 3.10.2)
project(sandhook LANGUAGES CXX)

# Force ARM64 architecture
if(NOT ANDROID_ABI STREQUAL "arm64-v8a")
    message(FATAL_ERROR "SandHook only supports arm64-v8a architecture.")
endif()

# Set C++ standard and flags
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Add the SandHook shared library (libsandhook.so)
add_library(sandhook SHARED
    src/sandhook.cpp
    src/art_hook.cpp
    src/sandhook_jni.cpp
)

# Include directories
target_include_directories(sandhook PRIVATE src)

# Compiler options for production (size optimized, no RTTI/Exceptions)
target_compile_options(sandhook PRIVATE
    -O2
    -fno-exceptions
    -fno-rtti
    -fPIC
    -Wall
    -Wextra
)

# Link required Android libraries
target_link_libraries(sandhook
    log    # For __android_log_print
    dl     # For dlopen/dlsym in art_hook.cpp
)
